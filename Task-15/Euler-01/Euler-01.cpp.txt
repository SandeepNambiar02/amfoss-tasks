int main(){
    int t,n;
    int count=0;
    cin >> t;
    for(int i=0;i<t;i++)
    {
        cin>>n;
        for(int j=1;j<n;j++)
        {
            if(j%3==0||j%5==0)
            {
                count+=j;
            }
        }
        cout<<count<<endl;
        count=0;
    }
    
    return 0;
}
